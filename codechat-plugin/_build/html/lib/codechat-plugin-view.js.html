  destroy() {
    if (!this.client.destroyed) {
      this.onCursor.dispose();
      this.onSave.dispose();
      this.client.destroy();
      this.onViewChange.dispose();
      this.onFileMod.dispose();

    console.log('destroyed');
    this.element.remove();
